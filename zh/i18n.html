    # i18n

    你可以自定义语言 json 数据显示定义的文案. 我们提供默认下面三种语言的支持 here.

    • en: English
    • zh-CN: 简体中文
    • jp: 日本語
    import VueCoreVideoPlayer from 'vue-core-video-player'

Vue.use(VueCoreVideoPlayer, {
  lang: 'zh-CN'
})

    Use custom language data:

    import VueCoreVideoPlayer from 'vue-core-video-player'

const kr = {
  ...
  "dashboard" : {
    "btn": {
      ....
      "pause": "일시적인",
      "fullscreen": "전체화면",
      "exitFullscreen": "전체 화면 종료",
    },
}


Vue.use(VueCoreVideoPlayer, {
  lang: kr
})
